<script>
    function login(tipo) {
        document.getElementById('login-screen').classList.add('hidden');
        document.getElementById(tipo + '-screen').classList.remove('hidden');
    }
    function logout() {
        document.querySelectorAll('.container > div').forEach(div => div.classList.add('hidden'));
        document.getElementById('login-screen').classList.remove('hidden');
    }
    function rate(stars) {
        let spans = document.querySelectorAll('#rating span');
        spans.forEach((s, i) => {
            s.classList.toggle('selected', i < stars);
        });
        document.getElementById('rating-result').innerText = "Você avaliou com " + stars + " estrelas!";
    }
    function marcarConcluida(btn) {
        btn.parentElement.style.textDecoration = "line-through";
        btn.disabled = true;
    }
</script>
